## Rule-Based Filtering Logic

### User Profile â†’ Service Filtering

```typescript
interface UserProfile {
  experienceLevel: 'beginner' | 'intermediate' | 'advanced'
  maintenancePreference: 'low' | 'moderate' | 'high'
  occasionContext: 'everyday' | 'professional' | 'special-occasion' | 'event'
  timeAvailable: 'quick' | 'standard' | 'extended'
  riskTolerance: 'minimal' | 'moderate' | 'higher'
}

// Example Filtering Rules
function filterServices(userProfile: UserProfile, allServices: Service[]) {
  return allServices.filter(service => {
    // Experience Level Filter
    if (userProfile.experienceLevel === 'beginner' && 
        !service.tags.includes('beginner-friendly')) {
      return false
    }
    
    // Maintenance Preference Filter  
    if (userProfile.maintenancePreference === 'low' && 
        service.tags.includes('high-maintenance')) {
      return false
    }
    
    // Occasion Context Filter
    if (userProfile.occasionContext === 'professional' && 
        !service.tags.includes('professional-setting') && 
        !service.tags.includes('everyday-appropriate')) {
      return false
    }
    
    // Risk Tolerance Filter
    if (userProfile.riskTolerance === 'minimal' && 
        service.tags.includes('higher-risk')) {
      return false
    }
    
    return true
  })
}
```

### Bundle Recommendation Logic

```typescript
function recommendBundles(selectedService: Service, allServices: Service[]) {
  const compatibleServices = allServices.filter(service => {
    // Don't bundle with self
    if (service.id === selectedService.id) return false
    
    // Find services with shared tags
    const sharedTags = selectedService.tags.filter(tag => 
      service.tags.includes(tag)
    )
    
    // Require at least 2 shared tags for bundling
    return sharedTags.length >= 2
  })
  
  // Prioritize by number of shared tags and complementary domains
  return compatibleServices.sort((a, b) => {
    const aSharedTags = selectedService.tags.filter(tag => a.tags.includes(tag)).length
    const bSharedTags = selectedService.tags.filter(tag => b.tags.includes(tag)).length
    
    // Bonus for cross-domain bundling
    const aCrossDomain = a.domain !== selectedService.domain ? 1 : 0
    const bCrossDomain = b.domain !== selectedService.domain ? 1 : 0
    
    return (bSharedTags + bCrossDomain) - (aSharedTags + aCrossDomain)
  })
}
```
